# VARIABLES REQUERIDAS

# Python version
PYTHON_VERSION=3.11.0

# Environment
ENVIRONMENT=production

# Secret key (GENERAR UNO NUEVO - ver instrucciones abajo)

# OpenAI API Key (para embeddings)

# Anthropic API Key (para query understanding)

# ============================================
# INSTRUCCIONES
# ============================================

# Para generar SECRET_KEY:
# Opción 1 (Python):
#   python3 -c "import secrets; print(secrets.token_urlsafe(32))"
#
# Opción 2 (OpenSSL):
#   openssl rand -base64 32
#
# Copiar el resultado y reemplazar en SECRET_KEY arriba

# Para obtener DATABASE_URL:
# 1. Ir a Render dashboard
# 2. Click en tu PostgreSQL database
# 3. Click "Info" tab
# 4. Copiar "External Database URL"
# 5. IMPORTANTE: Agregar ?sslmode=require al final
# 6. Pegar en DATABASE_URL arriba

# ============================================
# FORMATO PARA RENDER DASHBOARD
# ============================================

# En Render, agregar cada variable como:
# Key: PYTHON_VERSION
# Value: 3.11.0

# Key: SECRET_KEY
# Value: (tu key generado)
